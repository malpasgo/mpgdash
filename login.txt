    <script>
        // Konfigurasi password (dalam implementasi nyata, gunakan backend)
        const CORRECT_PASSWORD = "admin123"; // Ganti dengan password yang diinginkan
        
        // Elements
        const loginOverlay = document.getElementById('loginOverlay');
        const passwordInput = document.getElementById('passwordInput');
        const loginBtn = document.getElementById('loginBtn');
        const errorMessage = document.getElementById('errorMessage');
        const loadingAnimation = document.getElementById('loadingAnimation');
        const dashboardContainer = document.getElementById('dashboardContainer');
        const logoutBtn = document.getElementById('logoutBtn');

        // Check if user is already authenticated
        if (localStorage.getItem('authenticated') === 'true') {
            showDashboard();
        }

        // Login functionality
        loginBtn.addEventListener('click', handleLogin);
        passwordInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                handleLogin();
            }
        });

        function handleLogin() {
            const password = passwordInput.value;
            
            if (password === CORRECT_PASSWORD) {
                showLoading();
                
                // Simulate loading time
                setTimeout(() => {
                    localStorage.setItem('authenticated', 'true');
                    showDashboard();
                }, 1500);
            } else {
                showError();
                passwordInput.value = '';
                passwordInput.focus();
            }
        }

        function showError() {
            errorMessage.classList.add('show');
            passwordInput.style.borderColor = '#e74c3c';
            
            setTimeout(() => {
                errorMessage.classList.remove('show');
                passwordInput.style.borderColor = '#e1e5e9';
            }, 3000);
        }

        function showLoading() {
            loginOverlay.style.display = 'none';
            loadingAnimation.style.display = 'block';
        }

        function showDashboard() {
            loginOverlay.style.display = 'none';
            loadingAnimation.style.display = 'none';
            dashboardContainer.style.display = 'block';
        }

        // Logout functionality
        logoutBtn.addEventListener('click', function() {
            localStorage.removeItem('authenticated');
            dashboardContainer.style.display = 'none';
            loginOverlay.style.display = 'flex';
            passwordInput.value = '';
            passwordInput.focus();
        });

        // Focus on password input when page loads
        passwordInput.focus();

        // Prevent right-click and certain keyboard shortcuts for basic security
        document.addEventListener('contextmenu', function(e) {
            e.preventDefault();
        });

        document.addEventListener('keydown', function(e) {
            // Disable F12, Ctrl+Shift+I, Ctrl+U
            if (e.key === 'F12' || 
                (e.ctrlKey && e.shiftKey && e.key === 'I') ||
                (e.ctrlKey && e.key === 'u')) {
                e.preventDefault();
            }
        });
    </script>
